f\"🎬 Processing: {url}\\n\\n\"\n            \"Steps:\\n\"\n            \"1️⃣ Downloading from YouTube... ⏳\\n\"\n            \"2️⃣ Analyzing video... ⏳\\n\"\n            \"3️⃣ Detecting clips...     notification = f\"\"\"\n✅ **Processing Complete!**\n\n📊 Results:\n• Clips found: {clip_count}\n• Titles generated: {len(titles)}\n\n🎯 Generated Titles:\n\"\"\"\n        \n        for i, title in parse_mode='Markdown'\n        )\n    \n    def run(self):\n        \"\"\"Start bot\"\"\"\n        logger.info(\"Starting Telegram bot...\")\n        self.application.run_polling()\n\n\nif __name__ == '__main__':\n    bot = TelegramBotHandler()\n    bot.run()\n```\n\n### Step 4: Update .env\n\n```env\nTELEGRAM_BOT_TOKEN=your_bot_token_here\n```\n\n### Step 5: Run Bot\n\n```bash\npython Handler\n\nCreate file: `src/voice_handler.py`\n\n```python\n#!/usr/bin/env python\n\"\"\"\nVoice command handler for YouTube Auto-Clip Bot\n\"\"\"\n\nimport speech_recognition as sr\nimport pyttsx3\nimport logging\n\nlogger = logging.getLogger(__name__)\n\nclass VoiceCommandHandler:\n    \"\"\"Handles voice commands\"\"\"\n    \n    def __init__(self):\n        self.recognizer = sr.Recognizer()\n        self.engine = pyttsx3.init()\n        self.engine.setProperty('rate', 150)  # Speed\n        self.engine.setProperty('volume', 0.9)  # Volume\n    \n    def listen(self):\n        \"\"\"Listen for voice command\"\"\"\n        with sr.Microphone() as source:\n            logger.info(\"🎤 Listening...\")\n            print(\"Say command: (process, status, help, settings)\")\n            \n            try:\n                audio = self.recognizer.listen(source, timeout=10)\n                command = self.recognizer.recognize_google(audio)\n                logger.info(f\"Heard: {command}\")\n                return command.lower()\n            except sr.UnknownValueError:\n                self.speak(\"Sorry, I didn't understand. Please repeat.\")\n                return None\n            except sr.RequestError:\n                self.speak(\"Sorry, I'm having connection issues.\")\n                return None\n    \n    def speak(self, text):\n        \"\"\"Speak response\"\"\"\n        logger.info(f\"Speaking: {text}\")\n        self.engine.say(text)\n        self.engine.runAndWait()\n    \n    def process_command(self, command):\n        \"\"\"Process voice command\"\"\"\n        if not command:\n            return\n        \n        if 'process' in command or 'start' in command:\n            self.speak(\"Starting video processing. Please wait.\")\n            return 'process'\n        \n        elif 'status' in command:\n            self.speak(\"Checking bot status.\")\n            return 'status'\n        \n        elif 'help' in command:\n            self.speak(\"Here are available commands: process, status, settings, help.\")\n            return 'help'\n        \n        elif 'settings' in command:\n            self.speak(\"Opening settings.\")\n            return 'settings'\n        \n        elif 'stop' in command or 'exit' in command:\n            self.speak(\"Stopping bot. Goodbye!\")\n            return 'exit'\n        \n        else:\n            self.speak(\"Unknown command. Please try again.\")\n            return None\n    \n    def run(self):\n        \"\"\"Run voice command loop\"\"\"\n        self.speak(\"Voice commands activated.\")\n        \n        while True:\n            command = self.listen()\n            result = self.process_command(command)\n            \n            if result == 'exit':\n                break\n```\n\n### Step 3: Integrate with Main Bot\n\nUpdate `main.py`:\n\n```python\n# Add voice command support\nparser.add_argument(\n    '--voice',\n    action='store_true',\n    help='Enable voice commands'\n)\n\nif args.voice:\n    from src.voice_handler import VoiceCommandHandler\n    voice_handler = VoiceCommandHandler()\n    voice_handler.run()\n```\n\n---\n\n## 💬 Part 3: Discord Bot Integration\n\n### Step 1: Create Discord Bot\n\n1.
